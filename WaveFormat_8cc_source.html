44WaveFormat::WaveFormat( const std::string& file, const bool invert )
45{
46 const unsigned factor = invert ? -1 : 1;
47 std::string line;
48 std::ifstream fin( file, std::ios::in );
49
50 if( !fin.is_open() ){
51 usr::log::PrintLog( usr::log::FATAL,// Throws exception
52 usr::fstr( "Input file [%s] cannot be opened!", file ) );
53 }
54
55 // Getting first line
56 std::getline( fin, line );
57 std::istringstream linestream( line );
58 linestream >> time >> nbits >> adc;
59
60 // Getting all other lines
61 while( std::getline( fin, line ) ){
62 _waveforms.push_back( std::vector<int16_t>( line.length() / nbits ) );
63
64 for( unsigned index = 0; index < _waveforms.back().size(); ++index ){
65 int16_t value = 0;
66
67 for( unsigned bit = 0; bit < nbits; ++bit ){
68 const int16_t bit_value = hex_to_int( line[nbits * index + bit] );
69 value = value << 4 | bit_value;
70 }
71
72 _waveforms.back()[index] = nbits == 4 ? factor * value :
73 bit4_to_bit2( value * factor );
74 }
75
76 auto is_peak_cell
77 = [this]( const unsigned index )->bool {
78 const auto& w = this->_waveforms.back();
79 const unsigned diffp1 = index > w.size() - 2 ? 0 :
80 abs( w.at( index ) - w.at( index+1 ) );
81 const unsigned diffp2 = index > w.size() -3 ? 0 :
82 abs( w.at( index ) - w.at( index+2 ) );
83 const unsigned diffm1 = index < 1 ? 0 :
84 abs( w.at( index ) - w.at( index-1 ) );
85 const unsigned diffm2 = index < 2 ? 0 :
86 abs( w.at( index ) - w.at( index-2 ) );
87
88 const unsigned diffp = std::max( diffp1, diffp2 );
89 const unsigned diffm = std::max( diffm1, diffm2 );
90
91 return ( diffp > 70 ) && ( diffm > 70 );
92 };
93
94
95 for( unsigned index = 0; index < _waveforms.back().size(); ++index ){
96 if( is_peak_cell( index ) ){
97 _waveforms.back()[index]
98 = ( _waveforms.back()[index+1] + _waveforms.back()[index-1] ) / 2;
99 }
100 }
101 }
102}

130double
131WaveFormat::PedValue( const unsigned index,
132 const unsigned pedstart,
133 const unsigned pedstop ) const
134{
135 if( pedstart == unsigned(-1) || pedstop == unsigned(-1) ){
136 return 0;
137 }
138
139 double ped_value = 0;
140
141 for( unsigned i = pedstart; i < pedstop; ++i ){
142 ped_value += _waveforms.at( index ).at( i ) * ADC();
143 }
144
145 return ped_value / (double)( pedstop - pedstart );
146}

207{

210

214 for( unsigned i = start; i < stop; ++i ){
